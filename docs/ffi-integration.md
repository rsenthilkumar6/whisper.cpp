## Library Setup

### Loading the Dynamic Library

```dart
import 'dart:ffi';
import 'dart:io';

class WhisperBindings {
  late final DynamicLibrary _lib;

  WhisperBindings() {
    _lib = _loadLibrary();
  }

  DynamicLibrary _loadLibrary() {
    if (Platform.isMacOS) {
      // macOS: Try multiple locations
      final paths = [
        'libwhisper.dylib',
        '@executable_path/../Frameworks/libwhisper.dylib',
        '/usr/local/lib/libwhisper.dylib',
      ];
      for (final path in paths) {
        try {
          return DynamicLibrary.open(path);
        } catch (_) {}
      }
      throw Exception('Failed to load libwhisper.dylib');
    } else if (Platform.isLinux) {
      return DynamicLibrary.open('libwhisper.so');
    } else if (Platform.isWindows) {
      return DynamicLibrary.open('whisper.dll');
    } else if (Platform.isIOS) {
      return DynamicLibrary.process();  // Statically linked
    } else if (Platform.isAndroid) {
      return DynamicLibrary.open('libwhisper.so');
    }
    throw UnsupportedError('Platform not supported');
  }
}
```

```dart
typedef WhisperInitFromFileNative = Pointer<Void> Function(

  Pointer<Utf8> pathModel,

    .lookupFunction<WhisperInitFromFileNative, WhisperInitFromFile>(
        'whisper_init_from_file_with_params');

// Usage (IN ISOLATE)
Pointer<Void>? loadModel(String modelPath) {
  final pathPtr = modelPath.toNativeUtf8();
  try {
    final params = whisperContextDefaultParams();
    params.use_gpu = true;

    final ctx = whisperInitFromFileWithParams(pathPtr, params);
    if (ctx == nullptr) {
      throw Exception('Failed to load model: $modelPath');
    }
    return ctx;
  } finally {
    calloc.free(pathPtr);
  }
}
```

### whisper_init_state

```dart
// Dart FFI
typedef WhisperInitStateNative = Pointer<Void> Function(Pointer<Void> ctx);
typedef WhisperInitState = Pointer<Void> Function(Pointer<Void> ctx);

// Usage - multiple states for parallel processing
class ParallelWhisper {
  Pointer<Void>? _ctx;
  final List<Pointer<Void>> _states = [];

  void init(String modelPath, int numWorkers) {
    // Load model without state
    _ctx = whisperInitFromFileWithParamsNoState(modelPath, params);

    // Create states for each worker
    for (var i = 0; i < numWorkers; i++) {
      final state = whisperInitState(_ctx!);
      if (state == nullptr) throw Exception('Failed to create state $i');
      _states.add(state);
    }
  }

  void dispose() {
    for (final state in _states) {
      whisperFreeState(state);
    }
    if (_ctx != null) whisperFree(_ctx!);
  }
}
```

### whisper_full

Run complete transcription pipeline.

```c
// C API
int whisper_full(
    struct whisper_context * ctx,
    struct whisper_full_params params,
    const float * samples,
    int n_samples
);
// Returns 0 on success
```

```dart
// Dart FFI
typedef WhisperFullNative = Int32 Function(
  Pointer<Void> ctx,
  WhisperFullParams params,
  Pointer<Float> samples,
  Int32 nSamples,
);
typedef WhisperFull = int Function(
  Pointer<Void> ctx,
  WhisperFullParams params,
  Pointer<Float> samples,
  int nSamples,
);

// Usage (IN ISOLATE - this is the heavy operation)
int transcribe(Pointer<Void> ctx, List<double> audioSamples) {
  // Allocate native buffer
  final nSamples = audioSamples.length;
  final samplesPtr = calloc<Float>(nSamples);

  try {
    // Copy audio data
    for (var i = 0; i < nSamples; i++) {
      samplesPtr[i] = audioSamples[i];
    }

    // Get params
    final params = getTranscriptionParams(
      language: 'auto',
      threads: Platform.numberOfProcessors,
    );

    // Run transcription (blocking, CPU-intensive)
    final result = whisperFull(ctx, params, samplesPtr, nSamples);

    if (result != 0) {
      throw Exception('Transcription failed with code: $result');
    }

    return result;
  } finally {
    calloc.free(samplesPtr);
  }
}
```

```dart
// Usage for parallel transcription
Future<List<TranscriptionResult>> transcribeParallel(
  Pointer<Void> ctx,
  List<List<double>> audioChunks,
) async {
  final states = <Pointer<Void>>[];
  final futures = <Future<TranscriptionResult>>[];

  try {
    // Create state per chunk
    for (var i = 0; i < audioChunks.length; i++) {
      final state = whisperInitState(ctx);
      states.add(state);

      // Run in separate isolate
      futures.add(Isolate.run(() {
        return _transcribeChunk(ctx, state, audioChunks[i]);
      }));
    }

    return await Future.wait(futures);
  } finally {
    for (final state in states) {
      whisperFreeState(state);
    }
  }
}
```

```dart
// Dart FFI (IN ISOLATE)
class LanguageDetectionResult {
  final String language;
  final double confidence;
  final Map<String, double> allProbabilities;
}

LanguageDetectionResult detectLanguage(
  Pointer<Void> ctx,
  List<double> audioSamples,
) {
  // First, convert PCM to mel
  final samplesPtr = _toNativeFloatArray(audioSamples);
  whisperPcmToMel(ctx, samplesPtr, audioSamples.length, 4);
  calloc.free(samplesPtr);

  // Allocate probability array
  final maxLangId = whisperLangMaxId();
  final probsPtr = calloc<Float>(maxLangId + 1);

  try {
    final langId = whisperLangAutoDetect(ctx, 0, 4, probsPtr);

    if (langId < 0) {
      throw Exception('Language detection failed');
    }

    // Get all probabilities
    final probs = <String, double>{};
    for (var i = 0; i <= maxLangId; i++) {
      final lang = whisperLangStr(i).toDartString();
      probs[lang] = probsPtr[i];
    }

    return LanguageDetectionResult(
      language: whisperLangStr(langId).toDartString(),
      confidence: probsPtr[langId],
      allProbabilities: probs,
    );
  } finally {
    calloc.free(probsPtr);
  }
}
```

```dart
class SpeechSegment {

List<SpeechSegment> detectSpeechSegments(
  Pointer<Void> vadCtx,
  List<double> audio,
) {
  final params = whisperVadDefaultParams();
  params.threshold = 0.5;
  params.min_speech_duration_ms = 250;
  params.min_silence_duration_ms = 100;

  final samplesPtr = _toNativeFloatArray(audio);
  Pointer<Void>? segmentsPtr;

  try {
    segmentsPtr = whisperVadSegmentsFromSamples(
      vadCtx,
      params,
      samplesPtr,

    if (segmentsPtr == nullptr) {
      return [];
    }

    final nSegments = whisperVadSegmentsNSegments(segmentsPtr);

    for (var i = 0; i < nSegments; i++) {

        startSec: whisperVadSegmentsGetSegmentT0(segmentsPtr, i),
        endSec: whisperVadSegmentsGetSegmentT1(segmentsPtr, i),
      ));
    }

    return segments;
  } finally {
    calloc.free(samplesPtr);
    if (segmentsPtr != null) {
      whisperVadFreeSegments(segmentsPtr);
    }
  }
}
```

```dart
import 'dart:async';
import 'dart:ffi';
import 'dart:isolate';
import 'dart:typed_data';

class WhisperService {
  static WhisperService? _instance;
  late final SendPort _isolateSendPort;
  final _responseController = StreamController<WhisperResponse>.broadcast();

  Stream<WhisperResponse> get responses => _responseController.stream;

  static Future<WhisperService> getInstance() async {
    if (_instance == null) {
      _instance = WhisperService._();
      await _instance!._init();
    }
    return _instance!;
  }

  WhisperService._();

  Future<void> _init() async {
    final receivePort = ReceivePort();
    await Isolate.spawn(_isolateEntry, receivePort.sendPort);
    _isolateSendPort = await receivePort.first as SendPort;

    receivePort.listen((message) {
      if (message is WhisperResponse) {
        _responseController.add(message);
      }
    });
  }

  static void _isolateEntry(SendPort mainSendPort) {
    final receivePort = ReceivePort();
    mainSendPort.send(receivePort.sendPort);

    Pointer<Void>? ctx;
    final bindings = WhisperBindings();

    receivePort.listen((message) async {
      final request = message as WhisperRequest;

      switch (request.command) {
        case WhisperCommand.load:
          final path = request.data as String;
          ctx = bindings.loadModel(path);
          request.replyPort.send(WhisperResponse.loaded(ctx != null));
          break;

        case WhisperCommand.transcribe:
          if (ctx == null) {
            request.replyPort.send(
              WhisperResponse.error('Model not loaded'),
            );
            return;
          }

          final data = request.data as TranscribeRequest;
          try {
            final result = bindings.transcribe(
              ctx!,
              data.audio,
              language: data.language,
              translate: data.translate,
              onProgress: (p) {
                request.progressPort?.send(WhisperProgress(p));
              },
            );
            request.replyPort.send(WhisperResponse.transcription(result));
          } catch (e) {
            request.replyPort.send(WhisperResponse.error(e.toString()));
          }
          break;

        case WhisperCommand.dispose:
          if (ctx != null) {
            bindings.free(ctx!);
            ctx = null;
          }
          request.replyPort.send(WhisperResponse.disposed());
          break;
      }
    });
  }

  Future<bool> loadModel(String path) async {
    final response = await _sendCommand(
      WhisperCommand.load,
      path,
    );
    return response.success ?? false;
  }

  Future<TranscriptionResult> transcribe(
    Float32List audio, {
    String language = 'auto',
    bool translate = false,
    void Function(int progress)? onProgress,
  }) async {
    final progressPort = onProgress != null ? ReceivePort() : null;

    progressPort?.listen((msg) {
      if (msg is WhisperProgress) {
        onProgress(msg.percent);
      }
    });

    final response = await _sendCommand(
      WhisperCommand.transcribe,
      TranscribeRequest(
        audio: audio.toList(),
        language: language,
        translate: translate,
      ),
      progressPort: progressPort?.sendPort,
    );

    progressPort?.close();

    if (response.error != null) {
      throw Exception(response.error);
    }

    return response.result!;
  }

  Future<void> dispose() async {
    await _sendCommand(WhisperCommand.dispose, null);
    _responseController.close();
  }

  Future<WhisperResponse> _sendCommand(
    WhisperCommand command,
    dynamic data, {
    SendPort? progressPort,
  }) {
    final completer = Completer<WhisperResponse>();
    final responsePort = ReceivePort();

    responsePort.listen((msg) {
      completer.complete(msg as WhisperResponse);
      responsePort.close();
    });

    _isolateSendPort.send(WhisperRequest(
      command: command,
      data: data,
      replyPort: responsePort.sendPort,
      progressPort: progressPort,
    ));

    return completer.future;
  }
}

// Data classes
enum WhisperCommand { load, transcribe, dispose }

class WhisperRequest {
  final WhisperCommand command;
  final dynamic data;
  final SendPort replyPort;
  final SendPort? progressPort;

  WhisperRequest({
    required this.command,
    required this.data,
    required this.replyPort,
    this.progressPort,
  });
}

class WhisperResponse {
  final bool? success;
  final TranscriptionResult? result;
  final String? error;

  WhisperResponse.loaded(this.success) : result = null, error = null;
  WhisperResponse.transcription(this.result) : success = true, error = null;
  WhisperResponse.error(this.error) : success = false, result = null;
  WhisperResponse.disposed() : success = true, result = null, error = null;
}

class TranscribeRequest {
  final List<double> audio;
  final String language;
  final bool translate;

  TranscribeRequest({
    required this.audio,
    required this.language,
    required this.translate,
  });
}

class TranscriptionResult {
  final List<TranscriptionSegment> segments;
  final String fullText;
  final String? detectedLanguage;
  final Duration processingTime;

  TranscriptionResult({
    required this.segments,
    required this.fullText,
    this.detectedLanguage,
    required this.processingTime,
  });
}

class TranscriptionSegment {
  final String text;
  final Duration start;
  final Duration end;
  final List<TokenInfo>? tokens;

  TranscriptionSegment({
    required this.text,
    required this.start,
    required this.end,
    this.tokens,
  });
}

  final String text;

}
```

```dart
      _transcription = result.fullText;
    });

    _whisper?.dispose();
```
